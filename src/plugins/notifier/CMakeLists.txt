set(HEADERS
    notifier_plugin.h
    notifytextbrowser.h
    notifywidget.h
    animatedtextbrowser.h
    )

set(SOURCES
    notifier_plugin.cpp
    notifytextbrowser.cpp
    notifywidget.cpp
    animatedtextbrowser.cpp
    )

set(UIS
    notifywidget.ui
    )

include_directories(${CMAKE_CURRENT_BINARY_DIR} ".")

QT4_WRAP_CPP(MOCA_SOURCES ${HEADERS})
QT4_WRAP_CPP(UI_HEADERS_MOC ${UI_HEADERS})
QT4_WRAP_CPP(CORE_HEADERS_MOC ${CORE_HEADERS})
QT4_WRAP_UI(NOTIFIER_UI_HEADERS ${UIS})

add_library(notifierplugin SHARED
    ${SOURCES}
    ${MOCA_SOURCES}
    ${NOTIFIER_UI_HEADERS}
    )
	
target_link_libraries(notifierplugin
	${QT_LIBRARIES}
)

set_target_properties(notifierplugin
    PROPERTIES LIBRARY_OUTPUT_DIRECTORY ${PLUGIN_LIBRARY_OUTPUT_PATH}
    )

install(TARGETS notifierplugin 
	ARCHIVE DESTINATION ${libdir}
	LIBRARY DESTINATION ${libdir}
	COMPONENT library
)
